\subsubsection{最大团}
\begin{verbatim}
const int MAXN = 100; //number of vertices

int a[MAXN][MAXN];
int f[MAXN];
int N, ans;

bool DFS(int q[], int t, int cnt) {
    if (t == 0) {
        if (cnt > ans) {
            ans = cnt;
            return 1;
        }
        return 0;
    }

    int tq[MAXN];
    for (int i = 0; i < t; ++i) {
        if (f[q[i]] + cnt <= ans) return 0;
        int k = 0;
        for (int j = i + 1; j < t; ++j)
            if (a[q[i]][q[j]]) tq[k++] = q[j];
        if (DFS(tq, k, cnt + 1)) return 1;
    }
    return 0;
}

void MaxClique() {
    ans = 0;
    int q[MAXN];
    for (int i = N - 1; i >= 0; --i) {
        int t = 0;
        for (int j = i + 1; j < N; ++j) if (a[i][j]) q[t++] = j;
        DFS(q, t, 1);
        f[i] = ans;
    }
}
\end{verbatim}
